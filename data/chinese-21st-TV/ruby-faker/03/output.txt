class WuxiaFaker

  @@names = [

    '莫小贝', '郭芙蓉', '贞洁翠花', '华子鑫',

    '贾夫人', '母夜叉', '黑白无常', '杨大侠'
  ]

  def self.name
    @@names.sample
  end

end
