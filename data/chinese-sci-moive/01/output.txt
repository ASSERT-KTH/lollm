```python
import random

# 常见的中文姓氏
surnames = ["赵", "钱", "孙", "李", "周", "吴", "郑", "王", "冯", "陈", "楚", "卫"]

# 常见的中文名字
names = ["华", "宇", "林", "阳", "熙", "雨", "欣", "洋", "忠", "敏", "婷", "一", "国", "强"]

# 生成虚假的演员姓名
def generate_fake_name():
    surname = random.choice(surnames)
    name = "".join(random.sample(names, 2)) # 从names列表中随机选择两个字组成名字
    return surname + name

# 生成5个虚假的演员姓名
fake_actors = [generate_fake_name() for _ in range(5)]

# 打印生成的虚假演员姓名
for actor in fake_actors:
    print(actor)
```
